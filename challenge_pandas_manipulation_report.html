def get_summary_stats(df_train, df_test ):
    """
    :param df_train:
    :param df_test:
    """
    summary_stats_train = df_train.describe()
    missing_data_train = df_train.isnull().sum()
    print(f"Summary Statistics: {train_filename}\n {summary_stats_train}")
    print(f"\nMissing Data: \n {missing_data_train}")

    summary_stats_test = df_test.describe()
    missing_data_test = df_test.isnull().sum()
    print(f"\nSummary Statistics: {test_filename}\n {summary_stats_test}")
    print(f"\nMissing Data: \n {missing_data_train}")

def read_data(train_filename, test_filename, base_path = "./Part Two Data/", display_stats = False):
    """
    :param train_filename:
    :param test_filename:
    :param base_path:
    :param display_stats:
    """
    train_path = os.path.join(base_path,train_filename)
    test_path = os.path.join(base_path,test_filename)
    df_train = pd.read_csv(train_path)
    df_test = pd.read_csv(test_path)
    if display_stats:
        get_summary_stats(df_train, df_test )
    return df_train, df_test

In [9]:
# Normalize the values to the [0, 1] range
min_value_sensor_1 = min(df_full['sensor_1'])
max_value_sensor_1 = max(df_full['sensor_1'])
min_value_sensor_2 = min(df_full['sensor_2'])
max_value_sensor_2 = max(df_full['sensor_2'])
df_full['normalized_sensor_1'] = (df_full['sensor_1'] - min_value_sensor_1) / (max_value_sensor_1 - min_value_sensor_1)
df_full['normalized_sensor_2'] = (df_full['sensor_2'] - min_value_sensor_2) / (max_value_sensor_2 - min_value_sensor_2)

# Create a function to map normalized values to class labels
def map_to_class(value):
    for class_name, (start, end) in class_ranges.items():
        if start <= value <= end:
            return class_name
    return None  

df_full['Truth'] = df_full['normalized_sensor_1'].apply(map_to_class)
df_full['Predicted'] = df_full['normalized_sensor_2'].apply(map_to_class)

#clean up dataframe
df_full = df_full.drop(['normalized_sensor_1', 'normalized_sensor_2'], axis=1)
